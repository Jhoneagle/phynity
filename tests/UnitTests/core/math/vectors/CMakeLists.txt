phynity_target_defaults(vec2_test)

target_link_libraries(vec2_test PRIVATE phynity::tests Catch2::Catch2WithMain)
add_test(NAME unit.vectors.vec2 COMMAND ${CMAKE_COMMAND} -E env $<TARGET_FILE:vec2_test>)
